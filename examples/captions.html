Captions Demo

This demonstrates synchronized image captions (with lazyloading and offset value options). The wrapper div that is holding the container has been styled to have a transparent background, however it can be styled however you like. You can use simple text or complex HTML in your caption text, just remember to escape your quotes.

<div id="wrapper"></div>
<script src="//ajax.googleapis.com/ajax/libs/jquery/1.8.3/jquery.min.js"></script>
<script src="jquery.backstretch.min.js"></script>
<script>
    $.backstretch(
    [
      {
        "src" : "coffee.jpg",
        "caption" : "<h1>Coffee machine</h1><p>Can you smell the richness?</p>",
        "offsetX" : 1,
        "offsetY" : 0
      },
      {
        "src" : "cartesian.jpg",
        "caption" : "<h1>Cartesian</h1><p>Two dimensional grid.</p>",
        // "offsetX" : 1,
        // "offsetY" : 0
      },
      {
        "src" : "dome.jpg",
        // No caption!
        "offsetX" : 0,
        "offsetY" : 1
      },
      {
        "src" : "pot-holder.jpg",
        "caption" : "<h3>Pot Placemat</h3><p>Save your table cloth.</p>",
        "offsetX" : 0,
        "offsetY" : 1
      }
    ],
    {
        fade: 750,
        duration: 4000,
        lazyload: true,
        captionAppendTo: '#wrapper'
    });
</script>
